1 /**
  2  * @package pklib.css
  3  * @dependence pklib.string. pklib.dom
  4  */
  5 
  6 /**
  7  * Utils method related css on tags in DOM tree
  8  */
  9 pklib.css = (function () {
 10     "use strict";
 11 
 12     /**
 13      * RegExp use to delete white chars
 14      */
 15     var rclass = /[\n\t\r]/g;
 16 
 17     /**
 18      * Check typeof params
 19      *
 20      * @param {String} css_class
 21      * @param {HTMLElement} element
 22      * @throws {TypeError} If first param is not string, or second param is not Node
 23      */
 24     function check_params(css_class, element, call_func_name) {
 25         var prefix = "pklib.css." + call_func_name;
 26         pklib.common.assert(typeof css_class === "string", prefix + ": @css_class: not {String}");
 27         pklib.common.assert(pklib.dom.is_element(element), prefix + ": @element: not {HTMLElement}");
 28     }
 29 
 30     /**
 31      * Add CSS class to element define in second parameter
 32      *
 33      * @param {String} css_class
 34      * @param {HTMLElement} element
 35      * @throws {TypeError} If first param is not string, or second param is not Node
 36      */
 37     function add_class(css_class, element) {
 38         check_params(css_class, element, "add_class");
 39         var class_element = element.className;
 40         if (!pklib.css.has_class(css_class, element)) {
 41             if (class_element.length) {
 42                 class_element += " " + css_class;
 43             } else {
 44                 class_element = css_class;
 45             }
 46         }
 47         element.className = class_element;
 48     }
 49 
 50     /**
 51      * Remove CSS class from element define in second parameter
 52      *
 53      * @param {String} css_class
 54      * @param {HTMLElement} element
 55      * @throws {TypeError} If first param is not string, or second param is not Node
 56      */
 57     function remove_class(css_class, element) {
 58         check_params(css_class, element, "remove_class");
 59         var regexp = new RegExp("(\\s" + css_class + ")|(" + css_class + "\\s)|" + css_class, "i");
 60         element.className = pklib.string.trim(element.className.replace(regexp, ""));
 61     }
 62 
 63     /**
 64      * Check if element has CSS class
 65      *
 66      * @param {String} css_class
 67      * @param {HTMLElement} element
 68      * @throws {TypeError} If first param is not string, or second param is not Node
 69      * @returns {Boolean}
 70      */
 71     function has_class(css_class, element) {
 72         check_params(css_class, element, "has_class");
 73         var className = " " + css_class + " ";
 74         return ((" " + element.className + " ").replace(rclass, " ").indexOf(className) > -1);
 75     }
 76 
 77     // exports
 78     return {
 79         add_class: add_class,
 80         remove_class: remove_class,
 81         has_class: has_class
 82     };
 83 }());
