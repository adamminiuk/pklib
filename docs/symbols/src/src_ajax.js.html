1 /**
  2  * @package pklib.ajax
  3  * @dependence pklib.array, pklib.common
  4  */
  5 (function (global) {
  6     "use strict";
  7     /** @namespace */
  8     var pklib = global.pklib || {},
  9         /**
 10          * Default time what is timeout to use function pklib.ajax
 11          * @constant
 12          * @type Number
 13          */
 14         DEFAULT_TIMEOUT_TIME = 30000,
 15         /** @constant */
 16         REQUEST_STATE_UNSENT = 0,
 17         /** @constant */
 18         REQUEST_STATE_OPENED = 1,
 19         /** @constant */
 20         REQUEST_STATE_HEADERS_RECEIVED = 2,
 21         /** @constant */
 22         REQUEST_STATE_LOADING = 3,
 23         /** @constant */
 24         REQUEST_STATE_DONE = 4,
 25         /**
 26          * Array contain key as url, value as ajax response
 27          * @type Array
 28          */
 29         cache = [],
 30         /**
 31          * Use when state in request is changed or if used cache is handler to request.
 32          * @private
 33          * @function
 34          * @param {Object} settings
 35          * @param {XMLHttpRequest} xhr
 36          */
 37         handler = function (settings, xhr) {
 38             var contentType,
 39                 xmlContentType = ["application/xml", "text/xml"],
 40                 property = "responseText";
 41 
 42             if (xhr.readyState === REQUEST_STATE_DONE && xhr.status !== REQUEST_STATE_UNSENT) {
 43                 if (settings.cache) {
 44                     cache[settings.url] = xhr;
 45                 }
 46 
 47                 contentType = xhr.getResponseHeader("Content-Type");
 48 
 49                 if (pklib.array.inArray(contentType, xmlContentType)) {
 50                     property = "responseXML";
 51                 }
 52 
 53                 settings.done.call(null, xhr[property]);
 54 
 55                 // clear memory
 56                 xhr = null;
 57             }
 58         },
 59         /**
 60          * Handler to unusually situation - timeout.
 61          * @private
 62          * @function
 63          * @param {Object} settings
 64          * @param {XMLHttpRequest} xhr
 65          * @throws {Error} If exists timeout on request
 66          */
 67         timeoutHandler = function (settings, xhr) {
 68             // clear memory
 69             xhr = null;
 70             // throw exception
 71             throw new Error("pklib.ajax.load: timeout on url: " + settings.url);
 72         },
 73         /**
 74          * Method use when request has timeout
 75          * @private
 76          * @function
 77          * @param {Object} settings
 78          * @param {XMLHttpRequest} xhr
 79          * @throws {Error} If exists timeout on request
 80          */
 81         requestTimeout = function (settings, xhr) {
 82             if (typeof xhr.aborted === "undefined" &&
 83                     typeof xhr.error === "undefined" &&
 84                     xhr.readyState === REQUEST_STATE_DONE &&
 85                     xhr.status === REQUEST_STATE_UNSENT) {
 86                 xhr.abort();
 87                 timeoutHandler.call(null, settings, xhr);
 88             }
 89         },
 90         /**
 91          * Try to create Internet Explorer XMLHTTPRequest
 92          * @private
 93          * @function
 94          * @throws {Error} If can not create XMLHttpRequest object
 95          * @returns {Object|Undefined} ActiveXObject object
 96          */
 97         get_microsoft_xhr = function () {
 98             var xhr;
 99             try {
100                 xhr = new global.ActiveXObject("Msxml2.XMLHTTP");
101             } catch (ignore) {
102                 try {
103                     xhr = new global.ActiveXObject("Microsoft.XMLHTTP");
104                 } catch (ignored) {
105                     throw new Error("pklib.ajax.load: cannot create XMLHTTPRequest object");
106                 }
107             }
108             return xhr;
109         },
110         /**
111          * Try to create XMLHTTPRequest
112          * @private
113          * @function
114          * @throws {Error} If can not create XMLHttpRequest object
115          * @returns {Object|Undefined} XMLHttpRequest object
116          */
117         get_xhr = function () {
118             var xhr;
119             try {
120                 xhr = new global.XMLHttpRequest();
121             } catch (ignore) {
122                 xhr = get_microsoft_xhr();
123             }
124             return xhr;
125         },
126         /**
127          * Service to send request to server.
128          * With first param, which is hashmap, define params, ex. request url
129          * @namespace
130          */
131         ajax = {
132             /**
133              * Send request to server on url defined in config.url.
134              * Method throw exception when request have timeout on server or if url is not set.
135              * Also, every response (if config.cache is true) saved to hashmap by key config.url.
136              * Method on first try to can create XmlHttpRequest if browser doesn't support, check
137              * if browser support object ActiveXObject which is implemented in Internet Explorer.
138              * @memberOf ajax
139              * @function
140              * @param {Object} config
141              * <pre>
142              * {
143              *      {String} [type="get"]
144              *      {Boolean} [async=true]
145              *      {Boolean} [cache=false]
146              *      {String} url
147              *      {Object} [params]
148              *      {Object} [headers]
149              *      {Function} [done]
150              *      {Function} [error]
151              * }
152              * </pre>
153              * @example
154              * <pre>
155              * pklib.ajax.load({
156              *      type: "post",
157              *      async: false,
158              *      cache:  true,
159              *      url: "http://example.org/check-item.php",
160              *      params: {
161              *          id: 33
162              *      },
163              *      headers: {
164              *          "User-Agent": "tv"
165              *      },
166              *      done: function (res) {
167              *          console.log(res);
168              *      }
169              * });
170              * </pre>
171              * @throws {ReferenceError} If unset request url
172              * @returns {XMLHttpRequest|Null}
173              */
174             load: function (config) {
175                 var header,
176                     xhr = null,
177                     /**
178                      * Request settings, contain ex. headers, callback when run after request finish.
179                      * Default timeout on request is 30 seconds. This is default timeout from popular web servers
180                      * ex. Apache, ngninx.
181                      * Default request hasn't any headers.
182                      * Default cache is disabled.
183                      * Default asynchronous is enable.
184                      */
185                     settings = {
186                         type: "get",
187                         async: true,
188                         cache: false,
189                         url: null,
190                         params: null,
191                         timeout: DEFAULT_TIMEOUT_TIME,
192                         headers: {},
193                         /**
194                          * Function run after request ended
195                          */
196                         done: function (/* response */) {
197                             // do something with response
198                         },
199                         error: function () {
200                             // do something when appear error in request
201                         }
202                     };
203 
204                 settings = pklib.object.mixin(settings, config);
205                 settings.type = settings.type.toUpperCase();
206 
207                 if (settings.url === null) {
208                     throw new ReferenceError("pklib.ajax.load: undefined request url");
209                 }
210 
211                 if (settings.cache && cache[settings.url]) {
212                     handler.call(null, settings, cache[settings.url]);
213                 } else {
214                     xhr = get_xhr();
215                     xhr.onreadystatechange = handler.bind(null, settings, xhr);
216                     xhr.open(settings.type, settings.url, settings.async);
217 
218                     if (settings.headers !== null) {
219                         for (header in settings.headers) {
220                             if (settings.headers.hasOwnProperty(header)) {
221                                 xhr.setRequestHeader(header, settings.headers[header]);
222                             }
223                         }
224                     }
225                     xhr.send(settings.params);
226 
227                     if (typeof xhr.ontimeout === "function") {
228                         xhr.ontimeout = timeoutHandler.bind(null, settings, xhr);
229                     } else {
230                         pklib.common.defer(requestTimeout.bind(null, settings, xhr), settings.timeout);
231                     }
232 
233                     xhr.onerror = function () {
234                         xhr.error = true;
235                         settings.error.bind(null, settings, xhr);
236                     };
237                 }
238                 return xhr;
239             },
240             /**
241              * Stop request setting in param
242              * @memberOf ajax
243              * @function
244              * @param {XMLHttpRequest} xhr XMLHttpRequest object, or ActiveXObject object if Internet Explorer
245              */
246             stop: function (xhr) {
247                 xhr.abort();
248                 xhr.aborted = true;
249                 // clear memory
250                 xhr = null;
251             }
252         };
253 
254     pklib.ajax = ajax;
255 }(this));
