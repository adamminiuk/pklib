  6 /**
  7  * Service to send request to server.
  8  * With first param, which is hashmap, define params, ex. request url
  9  */
 10 (function (exports) {
 11     "use strict";
 12 
 13     var pklib = exports.pklib;
 14 
 15     // Default time what is timeout to use function pklib.ajax
 16     var DEFAULT_TIMEOUT_TIME = 30 * 1000; // 30 sekund
 17 
 18     var REQUEST_STATE_UNSENT = 0;
 19     var REQUEST_STATE_OPENED = 1;
 20     var REQUEST_STATE_HEADERS_RECEIVED = 2;
 21     var REQUEST_STATE_LOADING = 3;
 22     var REQUEST_STATE_DONE = 4;
 23 
 24     // Array contain key as url, value as ajax response
 25     var cache = [];
 26 
 27     /**
 28      * Use when state in request is changed or if used cache is handler to request
 29      *
 30      * @param {Object} settings
 31      * @param {XMLHttpRequest} xhr
 32      */
 33     function state_change_handler(settings, xhr) {
 34         var status = 0;
 35 
 36         if (xhr.readyState === REQUEST_STATE_DONE) {
 37             if (xhr.status !== undefined) {
 38                 status = xhr.status;
 39             }
 40 
 41             clearTimeout(xhr.ontimeout);
 42             delete xhr.ontimeout;
 43 
 44             if ((status >= 200 && status < 300) || status === 304) {
 45                 // success
 46                 success_handler(settings, xhr);
 47             } else {
 48                 // error
 49                 error_handler_with_abort(settings, xhr);
 50             }
 51         }
 52     }
 53 
 54     /**
 55      * When success request
 56      *
 57      * @param {Object} settings
 58      * @param {XMLHttpRequest} xhr
 59      */
 60     function success_handler(settings, xhr) {
 61         var contentType,
 62             xmlContentType = ["application/xml", "text/xml"],
 63             property = "responseText";
 64 
 65         if (settings.cache) {
 66             cache[settings.url] = xhr;
 67         }
 68 
 69         contentType = xhr.getResponseHeader("Content-Type");
 70 
 71         if (pklib.array.in_array(contentType, xmlContentType)) {
 72             property = "responseXML";
 73         }
 74 
 75         settings.done.call(null, xhr[property]);
 76 
 77         // clear memory
 78         xhr = null;
 79     }
 80 
 81     /**
 82      * When error request
 83      *
 84      * @param {Object} settings
 85      * @param {XMLHttpRequest} xhr
 86      */
 87     function error_handler(settings, xhr) {
 88         // check if error handler is run yet
 89         if (!xhr._run_error_handler) {
 90             // NO, so we run error handler first time
 91             settings.error(settings, xhr);
 92 
 93             // set flag to no run error handler
 94             xhr._run_error_handler = true;
 95         }
 96     }
 97 
 98     function error_handler_with_abort(settings, xhr) {
 99         xhr.abort();
100 
101         error_handler(settings, xhr);
102     }
103 
104     /**
105      * Handler to unusually situation - timeout
106      *
107      * @param {Object} settings
108      * @param {XMLHttpRequest} xhr
109      */
110     function timeout_handler(settings, xhr) {
111         error_handler(settings, xhr);
112     }
113 
114     /**
115      * Try to create Internet Explorer XMLHttpRequest
116      *
117      * @throws {Error} If can not create XMLHttpRequest object
118      * @returns {ActiveXObject|Undefined}
119      */
120     function create_microsoft_xhr() {
121         var xhr;
122         try {
123             xhr = new ActiveXObject("Msxml2.XMLHTTP");
124         } catch (ignore) {
125             try {
126                 xhr = new ActiveXObject("Microsoft.XMLHTTP");
127             } catch (ignored) {
128                 throw new Error("pklib.ajax.load: can't create XMLHttpRequest object");
129             }
130         }
131         return xhr;
132     }
133 
134     /**
135      * Try to create XMLHttpRequest
136      *
137      * @throws {Error} If can not create XMLHttpRequest object
138      * @returns {XMLHttpRequest|Undefined}
139      */
140     function create_xhr() {
141         var xhr;
142         try {
143             xhr = new XMLHttpRequest();
144         } catch (ignore) {
145             xhr = create_microsoft_xhr();
146         }
147         return xhr;
148     }
149 
150     /**
151      * Add headers to xhr object
152      *
153      * @param {Object} settings
154      * @param {XMLHttpRequest} xhr
155      */
156     function add_headers_to_xhr(settings, xhr) {
157         var header,
158             headers = settings.headers;
159 
160         if (headers !== null) {
161             for (header in headers) {
162                 if (headers.hasOwnProperty(header)) {
163                     xhr.setRequestHeader(header, headers[header]);
164                 }
165             }
166         }
167     }
168 
169     /**
170      * Add timeout service to xhr object
171      *
172      * @param {Object} settings
173      * @param {XMLHttpRequest} xhr
174      */
175     function add_timeout_service_to_xhr(settings, xhr) {
176         xhr.ontimeout = setTimeout(function () {
177             timeout_handler(settings, xhr);
178         }, settings.timeout);
179     }
180 
181     /**
182      * Check is response on this request is in cache
183      *
184      * @param {Object} settings
185      * @returns {Boolean}
186      */
187     function is_response_in_cache(settings) {
188         return cache[settings.url];
189     }
190 
191     /**
192      * Return object what is default configuration of request
193      *
194      * @returns {Object} Default configuration
195      */
196     function get_default_settings() {
197         /**
198          * Request settings, contain ex. headers, callback when run after request finish.
199          * Default timeout on request is 30 seconds. This is default timeout from popular web servers
200          * ex. Apache, ngninx.
201          * Default request hasn't any headers.
202          * Default cache is disabled.
203          * Default asynchronous is enable.
204          */
205         return {
206             type: "get",
207             async: true,
208             cache: false,
209             url: null,
210             params: null,
211             timeout: DEFAULT_TIMEOUT_TIME,
212             headers: {},
213             /**
214              * Function run after request ended
215              * In params exists only: response
216              */
217             done: function () {
218                 // do something when success request
219             },
220             error: function () {
221                 // do something when appear error in request
222             }
223         };
224     }
225 
226     // exports
227     pklib.ajax = {
228         /**
229          * Send request to server on url defined in config.url.
230          * Method throw exception when request have timeout on server or if url is not set.
231          * Also, every response (if config.cache is true) saved to hashmap by key config.url.
232          * Method on first try to can create XMLHttpRequest if browser doesn't support, check
233          * if browser support object ActiveXObject which is implemented in Internet Explorer.
234          *
235          * @param {Object} config
236          * <pre>
237          * {
238          *      {String} [type="get"]
239          *      {Boolean} [async=true]
240          *      {Boolean} [cache=false]
241          *      {String} url
242          *      {Object} [params]
243          *      {Object} [headers]
244          *      {Function} [done]
245          *      {Function} [error]
246          * }
247          * </pre>
248          * @example
249          * <pre>
250          * pklib.ajax.load({
251          *      type: "post",
252          *      async: false,
253          *      cache:  true,
254          *      url: "http://example.org/check-item.php",
255          *      params: {
256          *          id: 33
257          *      },
258          *      headers: {
259          *          "User-Agent": "tv"
260          *      },
261          *      done: function (res) {
262          *          // pass
263          *      }
264          * });
265          * </pre>
266          * @throws {Error} If unset request url
267          * @returns {XMLHttpRequest|Null}
268          */
269         load: function (config) {
270             var xhr = null,
271                 settings = get_default_settings();
272 
273             settings = pklib.object.mixin(settings, config);
274             settings.type = settings.type.toUpperCase();
275 
276             // simple assert to check "url" is set
277             pklib.common.assert(settings.url !== null, "pklib.ajax.load: @url is not defined");
278 
279             // check if we use "cache" flag in request
280             if (settings.cache && is_response_in_cache(settings)) {
281                 // YES, we use, so we can return response from cache object
282                 state_change_handler.call(null, settings, cache[settings.url]);
283             } else {
284                 // NO, is normal request to server
285                 xhr = create_xhr();
286                 xhr.onreadystatechange = state_change_handler.bind(null, settings, xhr);
287 
288                 try {
289                     xhr.open(settings.type, settings.url, settings.async);
290                 } catch (open_exception) {
291                     // error
292                     error_handler_with_abort(settings, xhr);
293 
294                     return xhr;
295                 }
296 
297                 add_headers_to_xhr(settings, xhr);
298                 add_timeout_service_to_xhr(settings, xhr);
299 
300                 try {
301                     xhr.send(settings.params);
302                 } catch (send_exception) {
303                     // error
304                     error_handler_with_abort(settings, xhr);
305 
306                     return xhr;
307                 }
308             }
309 
310             return xhr;
311         },
312 
313         /**
314          * Stop request setting in param
315          *
316          * @param {XMLHttpRequest|ActiveXObject} xhr XMLHttpRequest object, or ActiveXObject object if Internet Explorer
317          */
318         stop: function (xhr) {
319             xhr.abort();
320 
321             // clear memory
322             xhr = null;
323         }
324     };
325 }(this));
