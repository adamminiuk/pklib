1 /**
  2  * @package pklib.ajax
  3  * @dependence pklib.array, pklib.common

  6 /**
  7  * Service to send request to server.
  8  * With first param, which is hashmap, define params, ex. request url
  9  * @namespace
 10  */
 11 pklib.ajax = (function () {
 12     "use strict";
 13 
 14     var /**
 15          * Default time what is timeout to use function pklib.ajax
 16          *
 17          * @private
 18          * @constant
 19          * @type {Number}
 20          */
 21         DEFAULT_TIMEOUT_TIME = 30000,
 22 
 23         /**
 24          * @private
 25          * @constant
 26          * @type {Number}
 27          */
 28         REQUEST_STATE_UNSENT = 0,
 29 
 30         // REQUEST_STATE_OPENED = 1,
 31         // REQUEST_STATE_HEADERS_RECEIVED = 2,
 32         // REQUEST_STATE_LOADING = 3,
 33         /**
 34          * @private
 35          * @constant
 36          * @type {Number}
 37          */
 38         REQUEST_STATE_DONE = 4,
 39 
 40         /**
 41          * Array contain key as url, value as ajax response
 42          *
 43          * @private
 44          * @type {Array}
 45          */
 46         cache = [];
 47 
 48     /**
 49      * When success request
 50      *
 51      * @private
 52      * @function
 53      * @param {Object} settings
 54      * @param {XMLHttpRequest} xhr
 55      */
 56     function success_handler(settings, xhr) {
 57         var contentType,
 58             xmlContentType = ["application/xml", "text/xml"],
 59             property = "responseText";
 60 
 61         if (settings.cache) {
 62             cache[settings.url] = xhr;
 63         }
 64 
 65         contentType = xhr.getResponseHeader("Content-Type");
 66 
 67         if (pklib.array.in_array(contentType, xmlContentType)) {
 68             property = "responseXML";
 69         }
 70 
 71         settings.done.call(null, xhr[property]);
 72 
 73         // clear memory
 74         xhr = null;
 75     }
 76 
 77     /**
 78      * When error request
 79      *
 80      * @private
 81      * @function
 82      * @param {Object} settings
 83      * @param {XMLHttpRequest} xhr
 84      */
 85     function error_handler(settings, xhr) {
 86         xhr.error = true;
 87         settings.error.call(null, settings, xhr);
 88     }
 89 
 90     /**
 91      * Use when state in request is changed or if used cache is handler to request.
 92      *
 93      * @private
 94      * @function
 95      * @param {Object} settings
 96      * @param {XMLHttpRequest} xhr
 97      */
 98     function handler(settings, xhr) {
 99         var status = 0;
100 
101         if (xhr.readyState === REQUEST_STATE_DONE) {
102             if (xhr.status !== undefined) {
103                 status = xhr.status;
104             }
105 
106             if ((status >= 200 && status < 300) || status === 304) {
107                 // success
108                 success_handler(settings, xhr);
109             } else {
110                 // error
111                 error_handler(settings, xhr);
112             }
113         }
114     }
115 
116     /**
117      * Handler to unusually situation - timeout.
118      *
119      * @private
120      * @function
121      * @param {Object} settings
122      * @param {XMLHttpRequest} xhr
123      * @throws {Error} If exists timeout on request
124      */
125     function timeout_handler(settings, xhr) {
126         // clear memory
127         xhr = null;
128         // throw exception
129         throw new Error("pklib.ajax.load: timeout on url: " + settings.url);
130     }
131 
132     /**
133      * Method use when request has timeout
134      *
135      * @private
136      * @function
137      * @param {Object} settings
138      * @param {XMLHttpRequest} xhr
139      * @throws {Error} If exists timeout on request
140      */
141     function request_timeout(settings, xhr) {
142         if (xhr.aborted === undefined &&
143                 xhr.error === undefined &&
144                 xhr.readyState === REQUEST_STATE_DONE &&
145                 xhr.status === REQUEST_STATE_UNSENT) {
146             xhr.abort();
147             timeout_handler.call(null, settings, xhr);
148         }
149     }
150 
151     /**
152      * Try to create Internet Explorer XMLHttpRequest
153      *
154      * @private
155      * @function
156      * @throws {Error} If can not create XMLHttpRequest object
157      * @returns {ActiveXObject|Undefined}
158      */
159     function create_microsoft_xhr() {
160         var xhr;
161         try {
162             xhr = new ActiveXObject("Msxml2.XMLHTTP");
163         } catch (ignore) {
164             try {
165                 xhr = new ActiveXObject("Microsoft.XMLHTTP");
166             } catch (ignored) {
167                 throw new Error("pklib.ajax.load: can't create XMLHttpRequest object");
168             }
169         }
170         return xhr;
171     }
172 
173     /**
174      * Try to create XMLHttpRequest
175      *
176      * @private
177      * @function
178      * @throws {Error} If can not create XMLHttpRequest object
179      * @returns {XMLHttpRequest|Undefined}
180      */
181     function create_xhr() {
182         var xhr;
183         try {
184             xhr = new XMLHttpRequest();
185         } catch (ignore) {
186             xhr = create_microsoft_xhr();
187         }
188         return xhr;
189     }
190 
191     /**
192      * Add headers to xhr object
193      *
194      * @private
195      * @function
196      * @param {Object} settings
197      * @param {XMLHttpRequest} xhr
198      */
199     function add_headers_to_xhr(settings, xhr) {
200         var header,
201             headers = settings.headers;
202 
203         if (headers !== null) {
204             for (header in headers) {
205                 if (headers.hasOwnProperty(header)) {
206                     xhr.setRequestHeader(header, headers[header]);
207                 }
208             }
209         }
210     }
211 
212     /**
213      * Add timeout service to xhr object
214      *
215      * @private
216      * @function
217      * @param {Object} settings
218      * @param {XMLHttpRequest} xhr
219      */
220     function add_timeout_service_to_xhr(settings, xhr) {
221         if (typeof xhr.ontimeout === "function") {
222             xhr.ontimeout = timeout_handler.bind(null, settings, xhr);
223         } else {
224             pklib.common.defer(request_timeout.bind(null, settings, xhr), settings.timeout);
225         }
226     }
227 
228     /**
229      * Add error service to xhr object
230      *
231      * @private
232      * @function
233      * @param {Object} settings
234      * @param {XMLHttpRequest} xhr
235      */
236     function add_error_service_to_xhr(settings, xhr) {
237         xhr.onerror = function () {
238             error_handler(settings, xhr);
239         };
240     }
241 
242     /**
243      * Check is response on this request is in cache
244      *
245      * @private
246      * @function
247      * @param {Object} settings
248      * @returns {Boolean}
249      */
250     function is_response_in_cache(settings) {
251         return settings.cache && cache[settings.url];
252     }
253 
254     /**
255      * Return object what is default configuration of request
256      *
257      * @private
258      * @function
259      * @returns {Object} Default configuration
260      */
261     function get_default_settings() {
262         /**
263          * Request settings, contain ex. headers, callback when run after request finish.
264          * Default timeout on request is 30 seconds. This is default timeout from popular web servers
265          * ex. Apache, ngninx.
266          * Default request hasn't any headers.
267          * Default cache is disabled.
268          * Default asynchronous is enable.
269          */
270         return {
271             type: "get",
272             async: true,
273             cache: false,
274             url: null,
275             params: null,
276             timeout: DEFAULT_TIMEOUT_TIME,
277             headers: {},
278             /**
279              * Function run after request ended
280              * In params exists only: response
281              */
282             done: function () {
283                 // do something when success request
284             },
285             error: function () {
286                 // do something when appear error in request
287             }
288         };
289     }
290 
291     /**
292      * Check url in request is defined.
293      * Throw error if is undefined
294      *
295      * @private
296      * @function
297      * @param {Object} settings
298      * @throws {Error} If unset request url
299      */
300     function check_if_url_is_defined(settings) {
301         pklib.common.assert(settings.url !== null, "pklib.ajax.load: undefined request url");
302     }
303 
304     // public API
305     return {
306         /**
307          * Send request to server on url defined in config.url.
308          * Method throw exception when request have timeout on server or if url is not set.
309          * Also, every response (if config.cache is true) saved to hashmap by key config.url.
310          * Method on first try to can create XMLHttpRequest if browser doesn't support, check
311          * if browser support object ActiveXObject which is implemented in Internet Explorer.
312          *
313          * @memberOf pklib.ajax
314          * @function
315          * @param {Object} config
316          * <pre>
317          * {
318          *      {String} [type="get"]
319          *      {Boolean} [async=true]
320          *      {Boolean} [cache=false]
321          *      {String} url
322          *      {Object} [params]
323          *      {Object} [headers]
324          *      {Function} [done]
325          *      {Function} [error]
326          * }
327          * </pre>
328          * @example
329          * <pre>
330          * pklib.ajax.load({
331          *      type: "post",
332          *      async: false,
333          *      cache:  true,
334          *      url: "http://example.org/check-item.php",
335          *      params: {
336          *          id: 33
337          *      },
338          *      headers: {
339          *          "User-Agent": "tv"
340          *      },
341          *      done: function (res) {
342          *          // pass
343          *      }
344          * });
345          * </pre>
346          * @throws {Error} If unset request url
347          * @returns {XMLHttpRequest|Null}
348          */
349         load: function (config) {
350             var xhr = null,
351                 settings = get_default_settings();
352 
353             settings = pklib.object.mixin(settings, config);
354             settings.type = settings.type.toUpperCase();
355 
356             check_if_url_is_defined(settings);
357 
358             if (is_response_in_cache(settings)) {
359                 handler.call(null, settings, cache[settings.url]);
360             } else {
361                 xhr = create_xhr();
362                 xhr.onreadystatechange = handler.bind(null, settings, xhr);
363                 xhr.open(settings.type, settings.url, settings.async);
364 
365                 add_headers_to_xhr(settings, xhr);
366 
367                 add_timeout_service_to_xhr(settings, xhr);
368                 add_error_service_to_xhr(settings, xhr);
369                 xhr.send(settings.params);
370             }
371             return xhr;
372         },
373 
374         /**
375          * Stop request setting in param
376          *
377          * @memberOf pklib.ajax
378          * @function
379          * @param {XMLHttpRequest|ActiveXObject} xhr XMLHttpRequest object, or ActiveXObject object if Internet Explorer
380          */
381         stop: function (xhr) {
382             xhr.abort();
383             xhr.aborted = true;
384             // clear memory
385             xhr = null;
386         }
387     };
388 }());
