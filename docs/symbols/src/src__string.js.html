1 /**
  2  * @package pklib.string
  3  */
  4 (function (global) {
  5     "use strict";
  6 
  7     /**
  8      * @namespace
  9      * @type {Object}
 10      */
 11     var pklib = global.pklib || {},
 12         document = global.document || {};
 13 
 14     /**
 15      * String service manager
 16      * @namespace
 17      */
 18     pklib.string = {
 19         /**
 20          * @memberOf pklib.string
 21          * @function
 22          * @param {String} source
 23          * @returns {Boolean}
 24          */
 25         is_string: function (source) {
 26             return typeof source === "string";
 27         },
 28 
 29         /**
 30          * @memberOf pklib.string
 31          * @function
 32          * @param {String} source
 33          * @returns {Boolean}
 34          */
 35         is_letter: function (source) {
 36             return pklib.string.is_string(source) && /^[a-zA-Z]$/.test(source);
 37         },
 38 
 39         /**
 40          * @memberOf pklib.string
 41          * @function
 42          * @param {String} source
 43          * @returns {String}
 44          */
 45         trim: function (source) {
 46             return source.replace(/^\s+|\s+$/g, "");
 47         },
 48 
 49         /**
 50          * @memberOf pklib.string
 51          * @function
 52          * @param {String} source
 53          * @returns {String}
 54          */
 55         slug: function (source) {
 56             var result = source.toLowerCase().replace(/\s/mg, "-");
 57             result = result.replace(/[^a-zA-Z0-9\-]/mg, function (ch) {
 58                 switch (ch.charCodeAt(0)) {
 59                 case 261:
 60                     return String.fromCharCode(97);
 61                 case 281:
 62                     return String.fromCharCode(101);
 63                 case 243:
 64                     return String.fromCharCode(111);
 65                 case 347:
 66                     return String.fromCharCode(115);
 67                 case 322:
 68                     return String.fromCharCode(108);
 69                 case 378:
 70                 case 380:
 71                     return String.fromCharCode(122);
 72                 case 263:
 73                     return String.fromCharCode(99);
 74                 case 324:
 75                     return String.fromCharCode(110);
 76                 default:
 77                     return "";
 78                 }
 79             });
 80             return result;
 81         },
 82 
 83         /**
 84          * @memberOf pklib.string
 85          * @function
 86          * @param {String} source
 87          * @returns {String}
 88          */
 89         capitalize: function (source) {
 90             return source.substr(0, 1).toUpperCase() + source.substring(1, source.length).toLowerCase();
 91         },
 92 
 93         /**
 94          * @memberOf pklib.string
 95          * @function
 96          * @param {String} source
 97          * @returns {String}
 98          */
 99         delimiter_separated_words: function (source) {
100             return source.replace(/[A-ZĘÓĄŚŁŻŹĆŃ]/g, function (match) {
101                 return "-" + match.toLowerCase();
102             });
103         },
104 
105         /**
106          * @memberOf pklib.string
107          * @function
108          * @param {String} source
109          * @returns {String}
110          */
111         strip_tags: function (source) {
112             pklib.common.assert(typeof source === "string", "pklib.string.strip_tags: param @source is not a string");
113             if (source && source.length !== 0) {
114                 var dummy = document.createElement("div");
115                 dummy.innerHTML = source;
116                 return dummy.textContent || dummy.innerText;
117             }
118             return source;
119         },
120 
121         /**
122          * @memberOf pklib.string
123          * @function
124          * @param {String} source
125          * @returns {String}
126          */
127         camel_case: function (source) {
128             while (source.indexOf("-") !== -1) {
129                 var pos = source.indexOf("-"),
130                     pre = source.substr(0, pos),
131                     sub = source.substr(pos + 1, 1).toUpperCase(),
132                     post = source.substring(pos + 2, source.length);
133                 source = pre + sub + post;
134             }
135             return source;
136         },
137 
138         /**
139          * @memberOf pklib.string
140          * @function
141          * @param {String} source
142          * @param {Number} length
143          * @returns {String}
144          */
145         slice: function (source, length) {
146             var item,
147                 text = "",
148                 num = source.length;
149 
150             for (item = 0; item < num; item += 1) {
151                 text += source.substr(item, 1);
152                 if (item === length - 1) {
153                     if (num - length >= 1) {
154                         text += "...";
155                     }
156                     break;
157                 }
158             }
159             return text;
160         }
161     };
162 
163 }(this));
