1 /**
  2  * @package pklib.string
  3  */
  4 (function (global) {
  5     "use strict";
  6     /** @namespace */
  7     var pklib = global.pklib || {},
  8         document = global.document || {};
  9 
 10     /**
 11      * String service manager
 12      * @namespace
 13      */
 14     pklib.string = {
 15         /**
 16          * @memberOf pklib.string
 17          * @function
 18          * @param {String} source
 19          * @returns {Boolean}
 20          */
 21         is_string: function (source) {
 22             return typeof source === "string";
 23         },
 24         /**
 25          * @memberOf pklib.string
 26          * @function
 27          * @param {String} source
 28          * @returns {Boolean}
 29          */
 30         is_letter: function (source) {
 31             return pklib.string.is_string(source) && /^[a-zA-Z]$/.test(source);
 32         },
 33         /**
 34          * @memberOf pklib.string
 35          * @function
 36          * @param {String} source
 37          * @returns {String}
 38          */
 39         trim: function (source) {
 40             return source.replace(/^\s+|\s+$/g, "");
 41         },
 42         /**
 43          * @memberOf pklib.string
 44          * @function
 45          * @param {String} source
 46          * @returns {String}
 47          */
 48         slug: function (source) {
 49             var result = source.toLowerCase().replace(/\s/mg, "-");
 50             result = result.replace(/[^a-zA-Z0-9\-]/mg, function (ch) {
 51                 switch (ch.charCodeAt(0)) {
 52                 case 261:
 53                     return String.fromCharCode(97);
 54                 case 281:
 55                     return String.fromCharCode(101);
 56                 case 243:
 57                     return String.fromCharCode(111);
 58                 case 347:
 59                     return String.fromCharCode(115);
 60                 case 322:
 61                     return String.fromCharCode(108);
 62                 case 378:
 63                 case 380:
 64                     return String.fromCharCode(122);
 65                 case 263:
 66                     return String.fromCharCode(99);
 67                 case 324:
 68                     return String.fromCharCode(110);
 69                 default:
 70                     return "";
 71                 }
 72             });
 73             return result;
 74         },
 75         /**
 76          * @memberOf pklib.string
 77          * @function
 78          * @param {String} source
 79          * @returns {String}
 80          */
 81         capitalize: function (source) {
 82             return source.substr(0, 1).toUpperCase() + source.substring(1, source.length).toLowerCase();
 83         },
 84         /**
 85          * @memberOf pklib.string
 86          * @function
 87          * @param {String} source
 88          * @returns {String}
 89          */
 90         delimiter_separated_words: function (source) {
 91             return source.replace(/[A-ZĘÓĄŚŁŻŹĆŃ]/g, function (match) {
 92                 return "-" + match.toLowerCase();
 93             });
 94         },
 95         /**
 96          * @memberOf pklib.string
 97          * @function
 98          * @param {String} source
 99          * @returns {String}
100          */
101         strip_tags: function (source) {
102             pklib.common.assert(typeof source === "string", "pklib.string.strip_tags: param @source is not a string");
103             if (source && source.length !== 0) {
104                 var dummy = document.createElement("div");
105                 dummy.innerHTML = source;
106                 return dummy.textContent || dummy.innerText;
107             }
108             return source;
109         },
110         /**
111          * @memberOf pklib.string
112          * @function
113          * @param {String} source
114          * @returns {String}
115          */
116         camel_case: function (source) {
117             while (source.indexOf("-") !== -1) {
118                 var pos = source.indexOf("-"),
119                     pre = source.substr(0, pos),
120                     sub = source.substr(pos + 1, 1).toUpperCase(),
121                     post = source.substring(pos + 2, source.length);
122                 source = pre + sub + post;
123             }
124             return source;
125         },
126         /**
127          * @memberOf pklib.string
128          * @function
129          * @param {String} source
130          * @param {Number} length
131          * @returns {String}
132          */
133         slice: function (source, length) {
134             var item,
135                 text = "",
136                 num = source.length;
137 
138             for (item = 0; item < num; item += 1) {
139                 text += source.substr(item, 1);
140                 if (item === length - 1) {
141                     if (num - length >= 1) {
142                         text += "...";
143                     }
144                     break;
145                 }
146             }
147             return text;
148         }
149     };
150 }(this));
