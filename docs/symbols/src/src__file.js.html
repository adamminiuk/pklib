1 /**
  2  * @package pklib.file, pklib.string
  3  */
  4 (function (global) {
  5     "use strict";
  6     /** @namespace */
  7     var pklib = global.pklib || {},
  8         document = global.document || {},
  9         /**
 10          * @private
 11          * @type Array
 12          */
 13         copy_files = [],
 14         /**
 15          * @private
 16          * @function
 17          * @param {String} url
 18          * @param {Function} callback
 19          */
 20         simpleLoadJS = function (url, callback) {
 21             /**
 22              * Create HTMLElement <script>
 23              */
 24             var script = document.createElement("script");
 25             script.type = "text/javascript";
 26             script.src = url;
 27 
 28             if (typeof script.readyState === "undefined") {
 29                 /**
 30                  * Method run when request has ended
 31                  * @memberOf script
 32                  * @function
 33                  */
 34                 script.onload = function () {
 35                     if (typeof callback === "function") {
 36                         callback(script);
 37                     }
 38                 };
 39             } else {
 40                 /**
 41                  * Method run when request has change state
 42                  * @memberOf script
 43                  * @function
 44                  */
 45                 script.onreadystatechange = function () {
 46                     if (script.readyState === "loaded" || script.readyState === "complete") {
 47                         script.onreadystatechange = null;
 48                         if (typeof callback === "function") {
 49                             callback(script);
 50                         }
 51                     }
 52                 };
 53             }
 54 
 55             if (typeof document.head === "undefined") {
 56                 document.head = document.getElementsByTagName("head")[0];
 57             }
 58 
 59             document.head.appendChild(script);
 60         },
 61         /**
 62          * JS file loader
 63          * @namespace
 64          */
 65         file = {
 66             /**
 67              * Lazy load JS files. Url to files could be with path absolute or not.
 68              * If you must load more than 1 file use array, to set url to files
 69              * @memberOf file
 70              * @function
 71              * @param {String|Array} files
 72              * @param {Function} callback
 73              */
 74             loadjs: function (files, callback) {
 75                 var file,
 76                     self = this;
 77 
 78                 if (typeof files === "string") {
 79                     file = files;
 80                     simpleLoadJS(file, function (script) {
 81                         if (typeof callback === "function") {
 82                             callback(script);
 83                         }
 84                     });
 85                 } else if (pklib.array.is_array(files)) {
 86                     if (!copy_files.length) {
 87                         copy_files = pklib.object.mixin(copy_files, files);
 88                     }
 89 
 90                     file = files.shift();
 91 
 92                     if (typeof file === "undefined") {
 93                         if (typeof callback === "function") {
 94                             callback({
 95                                 src: copy_files[copy_files.length - 1]
 96                             });
 97 
 98                             copy_files = [];
 99                         }
100                     } else {
101                         simpleLoadJS(file, function () {
102                             self.loadjs(files, callback);
103                         });
104                     }
105                 } else {
106                     throw new TypeError("pklib.file.loadjs: @files not {String} or {Array}");
107                 }
108             }
109         };
110 
111     pklib.file = file;
112 }(this));
