1 /**
  2  * @package pklib.event
  3  */
  4 (function (global) {
  5     "use strict";
  6     /** @namespace */
  7     var pklib = global.pklib || {};
  8 
  9     /**
 10      * Helper about manage event on HTMLElement
 11      * @namespace
 12      */
 13     pklib.event = {
 14         /**
 15          * @memberOf pklib.event
 16          * @function
 17          * @param {HTMLElement} target
 18          * @param {String} event_name
 19          * @param {Function} handler
 20          */
 21         add: function (target, event_name, handler) {
 22             if (typeof target.events === "undefined") {
 23                 target.events = {};
 24             }
 25 
 26             var event = target.events[event_name];
 27 
 28             if (typeof event === "undefined") {
 29                 target.events[event_name] = [];
 30             }
 31 
 32             target.events[event_name].push(handler);
 33 
 34             if (target.attachEvent) {
 35                 // IE browser
 36                 target.attachEvent("on" + event_name, handler);
 37             } else if (target.addEventListener) {
 38                 // other browser
 39                 target.addEventListener(event_name, handler, false);
 40             } else {
 41                 // for very old browser
 42                 target["on" + event_name] = handler;
 43             }
 44         },
 45         /**
 46          * @memberOf pklib.event
 47          * @function
 48          * @param {HTMLElement} target
 49          * @param {String} event_name
 50          */
 51         remove: function (target, event_name) {
 52             var removeEvent, events, len, i, handler;
 53 
 54             if (typeof target.events === "undefined") {
 55                 target.events = {};
 56             }
 57 
 58             if (target.detachEvent) {
 59                 // IE browser
 60                 removeEvent = "detachEvent";
 61             } else if (target.removeEventListener) {
 62                 // other browser
 63                 removeEvent = "removeEventListener";
 64             }
 65 
 66             if (typeof removeEvent === "undefined") {
 67                 // for old browser
 68                 delete target["on" + event_name];
 69             } else {
 70                 events = target.events[event_name];
 71 
 72                 if (typeof events !== "undefined") {
 73                     len = events.length;
 74 
 75                     for (i = 0; i < len; ++i) {
 76                         handler = events[i];
 77                         target[removeEvent](event_name, handler);
 78                         delete target.events[event_name];
 79                     }
 80                 }
 81             }
 82         },
 83         /**
 84          * @memberOf pklib.event
 85          * @function
 86          * @param {HTMLElement} target
 87          * @param {String} event_name
 88          * @returns {Array|Undefined}
 89          */
 90         get: function (target, event_name) {
 91             if (typeof target.events === "undefined") {
 92                 target.events = {};
 93             }
 94             return target.events[event_name];
 95         },
 96         /**
 97          * @memberOf pklib.event
 98          * @function
 99          * @param {HTMLElement} target
100          * @param {String} event_name
101          * @throws {ReferenceError} If HTMLElement haven't got any events
102          */
103         trigger: function (target, event_name) {
104             var events, len, i;
105 
106             if (typeof target.events === "undefined") {
107                 target.events = {};
108             }
109 
110             events = target.events[event_name];
111 
112             pklib.common.assert(typeof events !== "undefined", "pklib.event.trigger: @event " + event_name + ": not {Array}");
113 
114             len = events.length;
115 
116             for (i = 0; i < len; ++i) {
117                 events[i].call(target, events[i]);
118             }
119         }
120     };
121 }(this));
