1 /**
  2  * @package pklib.css
  3  * @dependence pklib.string. pklib.dom
  4  */
  5 (function (global) {
  6     "use strict";
  7 
  8     /**
  9      * @namespace
 10      * @type {Object}
 11      */
 12     var pklib = global.pklib || {},
 13         /**
 14          * RegExp use to delete white chars
 15          * @private
 16          * @type {RegExp}
 17          */
 18         rclass = /[\n\t\r]/g,
 19 
 20         /**
 21          * Check typeof params
 22          * @private
 23          * @function
 24          * @param {String} css_class
 25          * @param {HTMLElement} element
 26          * @throws {TypeError} If first param is not string, or second param is not Node
 27          */
 28         check_params = function (css_class, element, call_func_name) {
 29             var prefix = "pklib.css." + call_func_name;
 30             pklib.common.assert(typeof css_class === "string", prefix + ": @css_class: not {String}");
 31             pklib.common.assert(pklib.dom.is_element(element), prefix + ": @element: not {HTMLElement}");
 32         };
 33 
 34     /**
 35      * Utils method related css on tags in DOM tree
 36      * @namespace
 37      */
 38     pklib.css = {
 39         /**
 40          * Add CSS class to element define in second parameter
 41          * @memberOf pklib.css
 42          * @function
 43          * @param {String} css_class
 44          * @param {HTMLElement} element
 45          * @throws {TypeError} If first param is not string, or second param is not Node
 46          */
 47         add_class: function (css_class, element) {
 48             check_params(css_class, element, "add_class");
 49             var class_element = element.className;
 50             if (!pklib.css.has_class(css_class, element)) {
 51                 if (class_element.length) {
 52                     class_element += " " + css_class;
 53                 } else {
 54                     class_element = css_class;
 55                 }
 56             }
 57             element.className = class_element;
 58         },
 59 
 60         /**
 61          * Remove CSS class from element define in second parameter
 62          * @memberOf pklib.css
 63          * @function
 64          * @param {String} css_class
 65          * @param {HTMLElement} element
 66          * @throws {TypeError} If first param is not string, or second param is not Node
 67          */
 68         remove_class: function (css_class, element) {
 69             check_params(css_class, element, "remove_class");
 70             var regexp = new RegExp("(\\s" + css_class + ")|(" + css_class + "\\s)|" + css_class, "i");
 71             element.className = pklib.string.trim(element.className.replace(regexp, ""));
 72         },
 73 
 74         /**
 75          * Check if element has CSS class
 76          * @memberOf pklib.css
 77          * @function
 78          * @param {String} css_class
 79          * @param {HTMLElement} element
 80          * @throws {TypeError} If first param is not string, or second param is not Node
 81          * @returns {Boolean}
 82          */
 83         has_class: function (css_class, element) {
 84             check_params(css_class, element, "has_class");
 85             var className = " " + css_class + " ";
 86             return ((" " + element.className + " ").replace(rclass, " ").indexOf(className) > -1);
 87         }
 88     };
 89 
 90 }(this));
