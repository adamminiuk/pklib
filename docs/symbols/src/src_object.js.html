1 /**
  2  * @package pklib.object
  3  */
  4 
  5 /**
  6  * Module to service object
  7  */
  8 pklib.object = (function () {
  9     "use strict";
 10 
 11     var to_string = Object.prototype.toString;
 12 
 13     /**
 14      * Check if param is object
 15      *
 16      * @param {Object} obj
 17      * @returns {Boolean}
 18      */
 19     function is_object(obj) {
 20         return obj &&
 21             to_string.call(obj) === "[object Object]" &&
 22             typeof obj === "object" &&
 23             typeof obj.hasOwnProperty === "function" &&
 24             typeof obj.isPrototypeOf === "function";
 25     }
 26 
 27     /**
 28      * Mix two params, from second to first param. Return first param mixin with second param
 29      *
 30      * @param {Array|Object} target
 31      * @param {Array|Object} source
 32      * @returns {Array|Object}
 33      */
 34     function mixin(target, source) {
 35         var i, len, element, item;
 36 
 37         if (pklib.array.is_array(target) && pklib.array.is_array(source)) {
 38             len = source.length;
 39 
 40             for (i = 0; i < len; ++i) {
 41                 element = source[i];
 42 
 43                 if ( !pklib.array.in_array(element, target) ) {
 44                     target.push(element);
 45                 }
 46             }
 47         } else {
 48             for (item in source) {
 49                 if (source.hasOwnProperty(item)) {
 50                     if (pklib.object.is_object(target[item])) {
 51                         target[item] = pklib.object.mixin(target[item], source[item]);
 52                     } else {
 53                         target[item] = source[item];
 54                     }
 55                 }
 56             }
 57         }
 58         return target;
 59     }
 60 
 61     // exports
 62     return {
 63         is_object: is_object,
 64         mixin: mixin
 65     };
 66 }());
