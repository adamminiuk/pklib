  5 /**
  6  * Bind function to aspect.
  7  * Create method with merge first and second.
  8  * Second method is run after first
  9  *
 10  * @function
 11  * @param {Function} fun The function to bind aspect function
 12  * @param {Function} asp The aspect function
 13  * @param {String} [when] Place to aspect function
 14  * @throws {TypeError} If any param is not function
 15  * @returns {Function}
 16  */
 17 pklib.aspect = function (fun, asp, when) {
 18     "use strict";
 19 
 20     var that = this,
 21         result;
 22 
 23     pklib.common.assert(typeof fun === "function", "pklib.aspect: @func: not {Function}");
 24     pklib.common.assert(typeof asp === "function", "pklib.aspect: @asp: not {Function}");
 25 
 26     when = when || "before";
 27 
 28     return function () {
 29         if (when === "before") {
 30             asp.call(that);
 31         }
 32 
 33         result = fun.apply(that, arguments);
 34 
 35         if (when === "after") {
 36             result = asp.call(that);
 37         }
 38         return result;
 39     };
 40 };
