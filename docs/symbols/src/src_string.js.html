  8 pklib.string = (function () {
  9     "use strict";

 11     /**
 12      * @param {String} source
 13      * @returns {Boolean}
 14      */
 15     function is_string(source) {
 16         return typeof source === "string";
 17     }
 18 
 19     /**
 20      * @param {String} source
 21      * @returns {Boolean}
 22      */
 23     function is_letter(source) {
 24         return pklib.string.is_string(source) && /^[a-zA-Z]$/.test(source);
 25     }
 26 
 27     /**
 28      * @param {String} source
 29      * @returns {String}
 30      */
 31     function trim(source) {
 32         return source.replace(/^\s+|\s+$/g, "");
 33     }
 34 
 35     /**
 36      * @param {String} source
 37      * @returns {String}
 38      */
 39     function slug(source) {
 40         var result = source.toLowerCase().replace(/\s/mg, "-");
 41         result = result.replace(/[^a-zA-Z0-9\-]/mg, function (ch) {
 42             switch (ch.charCodeAt(0)) {
 43             case 261:
 44                 return String.fromCharCode(97);
 45             case 281:
 46                 return String.fromCharCode(101);
 47             case 243:
 48                 return String.fromCharCode(111);
 49             case 347:
 50                 return String.fromCharCode(115);
 51             case 322:
 52                 return String.fromCharCode(108);
 53             case 378:
 54             case 380:
 55                 return String.fromCharCode(122);
 56             case 263:
 57                 return String.fromCharCode(99);
 58             case 324:
 59                 return String.fromCharCode(110);
 60             default:
 61                 return "";
 62             }
 63         });
 64         return result;
 65     }

 67     /**
 68      * @param {String} source
 69      * @returns {String}
 70      */
 71     function capitalize(source) {
 72         return source.substr(0, 1).toUpperCase() + source.substring(1, source.length).toLowerCase();
 73     }
 74 
 75     /**
 76      * @param {String} source
 77      * @returns {String}
 78      */
 79     function delimiter_separated_words(source) {
 80         return source.replace(/[A-ZĘÓĄŚŁŻŹĆŃ]/g, function (match) {
 81             return "-" + match.toLowerCase();
 82         });
 83     }
 84 
 85     /**
 86      * @param {String} source
 87      * @returns {String}
 88      */
 89     function strip_tags(source) {
 90         pklib.common.assert(typeof source === "string", "pklib.string.strip_tags: param @source is not a string");
 91         if (source && source.length !== 0) {
 92             var dummy = document.createElement("div");
 93             dummy.innerHTML = source;
 94             return dummy.textContent || dummy.innerText;
 95         }
 96         return source;
 97     }
 98 
 99     /**
100      * @param {String} source
101      * @returns {String}
102      */
103     function camel_case(source) {
104         while (source.indexOf("-") !== -1) {
105             var pos = source.indexOf("-"),
106                 pre = source.substr(0, pos),
107                 sub = source.substr(pos + 1, 1).toUpperCase(),
108                 post = source.substring(pos + 2, source.length);
109             source = pre + sub + post;
110         }
111         return source;
112     }
113 
114     /**
115      * @param {String} source Text to slice
116      * @param {Number} length Number of chars what string will be slice
117      * @param {Boolean} [is_force] Force mode. If slice will be end in middle of word, use this to save it, or algorytm slice to last space
118      * @returns {String}
119      */
120     function slice(source, length, is_force) {
121         pklib.common.assert(typeof source === "string", "pklib.string.slice: param @source is not a string");
122 
123         // jeśli długość przycinania jest wiąksza niż długość całego tekstu
124         // to zwracamy przekazany tekst
125         if (source.length < length) {
126             return source;
127         }
128 
129         // ucinamy tyle tekstu ile jest wskazane w parametrze length
130         var text = source.slice(0, length), last_space;
131 
132         // sprawdzamy czy nie ucieliśmy w połowie wyrazu:
133         // * tj. czy kolejnym znakiem nie jest spacja
134         if (source[length] === " ") {
135             return text + "...";
136         }
137 
138         // * ostatnim znakiem w uciętym tekście jest spacja
139         if (text[length - 1] === " ") {
140             return pklib.string.trim(text) + "...";
141         }
142 
143         // jesli nie ma wymuszenia przycinania wyrazu w jego części 
144         // to sprawdzamy czy możemy przyciąć do ostatniej spacji w przycietym tekście
145         if (!is_force) {
146             // niestety ucieliśmy tekst w połowie wyrazu
147             // postępujemy zgodnie z intrukcja, że odnajdujemy ostatnią spację
148             // i obcinamy fraze do tej spacji 
149             last_space = text.lastIndexOf(" ");
150 
151             // spacja została znaleziona, więc przycinamy frazę do spacji
152             if (last_space !== -1) {
153                 return text.slice(0, last_space) + "...";
154             }
155         }
156 
157         // włączony tryb "force" albo spacja nie została odnaleziona więc aby nie zwracać 
158         // w pustej wartości, ucinamy wyraz w tym miejscu w którym jest
159         return text + "...";
160     }

162     /**
163      * Replace tags in string to defined data
164      * Tags:
165      * ${NAME} - replace by value of object["NAME"]
166      *
167      * @param {String} str Some string to replace by objects
168      * @param {Object} obj Object what will serve data to replacer
169      *
170      * @example
171      *
172      * In: 
173      * %{car} is the best!
174      *
175      * Run: 
176      * pklib.string.format("%{car} is the best", { car: "Ferrari" });
177      *
178      * Out: 
179      * Ferrari is the best!
180      */
181     function format(str, obj) {
182         var name;
183 
184         for (name in obj) {
185             if (obj.hasOwnProperty(name)) {
186                 str = str.replace(new RegExp("%{" + name + "}", "ig"), obj[name]);
187             }
188         }
189 
190         return str;
191     }
192 
193     /**
194      * Left padding any chars
195      *
196      * @param {String} staff Object what be padding on the left
197      * @param {Number} nr_fill Padding size
198      * @param {String} add_char Char what be added
199      */
200     function lpad(staff, nr_fill, add_char) {
201         var string = staff.toString(),
202             i = string.length;
203 
204         for (; i < nr_fill; ++i) {
205             string = add_char + string;
206         }
207 
208         return string;
209     }
210 
211     /**
212      * Right padding any chars
213      *
214      * @param {String} staff Object what be padding on the right
215      * @param {Number} nr_fill Padding size
216      * @param {String} add_char Char what be added
217      */
218     function rpad(staff, nr_fill, add_char) {
219         var string = staff.toString(),
220             i = string.length;
221 
222         for (; i < nr_fill; ++i) {
223             string += add_char;
224         }
225 
226         return string;
227     }

237         strip_tags: strip_tags,
238         camel_case: camel_case,
239         slice: slice,
240         format: format,
241         lpad: lpad,
242         rpad: rpad
243     };
