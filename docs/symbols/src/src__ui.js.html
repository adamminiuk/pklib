1 /**
  2  * @package pklib.ui
  3  * @dependence pklib.string. pklib.dom
  4  */
  5 (function (global) {
  6     "use strict";
  7 
  8     /**
  9      * @namespace
 10      * @type {Object}
 11      */
 12     var pklib = global.pklib || {},
 13         document = global.document || {};
 14 
 15     /**
 16      * User Interface
 17      * @namespace
 18      */
 19     pklib.ui = {
 20         /**
 21          * @memberOf pklib.ui
 22          * @function
 23          * @param {HTMLElement} element
 24          * @param {HTMLElement} wrapper
 25          * @throws {TypeError} If first param is not HTMLElement
 26          * @returns {Array}
 27          */
 28         center: function (element, wrapper) {
 29             var left = null,
 30                 top = null,
 31                 pus = this.size;
 32 
 33             pklib.common.assert(pklib.dom.is_element(element), "pklib.ui.center: @element: not {HTMLElement}");
 34 
 35             if (wrapper === document.body) {
 36                 left = (Math.max(pus.window("width"), pus.document("width")) - pus.object(element, "width")) / 2;
 37                 top = (Math.max(pus.window("height"), pus.document("height")) - pus.object(element, "height")) / 2;
 38             } else {
 39                 left = (pus.window("width") - pus.object(element, "width")) / 2;
 40                 top = (pus.window("height") - pus.object(element, "height")) / 2;
 41             }
 42             element.style.left = left + "px";
 43             element.style.top = top + "px";
 44             element.style.position = "absolute";
 45 
 46             return [left, top];
 47         },
 48         /**
 49          * @memberOf pklib.ui
 50          * @function
 51          * @param {HTMLElement} element
 52          * @param {HTMLElement} wrapper
 53          * @returns {Array}
 54          */
 55         maximize: function (element, wrapper) {
 56             var width = null,
 57                 height = null,
 58                 pus = pklib.ui.size;
 59 
 60             if (wrapper === document.body) {
 61                 width = Math.max(pus.window("width"), pus.document("width"));
 62                 height = Math.max(pus.window("height"), pus.document("height"));
 63                 if (pklib.browser.get_name() === "msie") {
 64                     width -= 20;
 65                 }
 66             } else {
 67                 width = pus.object(wrapper, "width");
 68                 height = pus.object(wrapper, "height");
 69             }
 70             element.style.width = width;
 71             element.style.height = height;
 72             return [width, height];
 73         },
 74         /**
 75          * @memberOf pklib.ui
 76          * @function
 77          * @param {Number} param
 78          * @param {Boolean} animate
 79          */
 80         scroll_to: function (param, animate) {
 81             var interval = null;
 82             if (animate) {
 83                 interval = setInterval(function () {
 84                     document.body.scroll_top -= 5;
 85                     if (document.body.scroll_top <= 0) {
 86                         clearInterval(interval);
 87                     }
 88                 }, 1);
 89             } else {
 90                 document.body.scroll_top = param + "px";
 91             }
 92         }
 93     };
 94 
 95 }(this));
