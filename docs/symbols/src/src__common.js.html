1 /**
  2  * @package pklib.common
  3  */
  4 (function (global) {
  5     "use strict";
  6 
  7     /**
  8      * @namespace
  9      * @type {Object}
 10      */
 11     var pklib = global.pklib || {};
 12 
 13     /**
 14      * Common stuff
 15      * @namespace
 16      */
 17     pklib.common = {
 18         /**
 19          * Basic test function. Simple assertion 2 variables
 20          * @memberOf pklib.common
 21          * @function
 22          * @param {Object} expression Object what is true
 23          * @param {String} comment Message to throw in error
 24          * @throws {Error}
 25          */
 26         assert: function (expression, comment) {
 27             if (expression !== true) {
 28                 throw new Error(comment);
 29             }
 30         },
 31 
 32         /**
 33          * Deferred function about some milliseconds.
 34          * If milliseconds is 0 that it's hack for some platforms to use function in "next" thread
 35          * @memberOf pklib.common
 36          * @function
 37          * @param {Function} defer_function Function what would be deferred
 38          * @param {Number} milliseconds Time to deferred function
 39          */
 40         defer: function (defer_function, milliseconds) {
 41             milliseconds = milliseconds || 0;
 42             setTimeout(defer_function, milliseconds);
 43         },
 44 
 45         /**
 46          * Interval checking first function until returns true,
 47          * run after this second function callback
 48          * @memberOf pklib.common
 49          * @param {Function} condition Function returns {Boolean} status
 50          * @param {Function} callback
 51          */
 52         checking: function (condition, callback) {
 53             var interval,
 54                 interval_time = 100;
 55 
 56             pklib.common.assert(typeof condition === "function", "pklib.common.checking: @condition: not {Function}");
 57             pklib.common.assert(typeof callback === "function", "pklib.common.checking: @callback: not {Function}");
 58 
 59             if (condition()) {
 60                 callback();
 61             } else {
 62                 interval = setInterval(function () {
 63                     if (condition()) {
 64                         clearInterval(interval);
 65                         callback();
 66                     }
 67                 }, interval_time);
 68             }
 69         }
 70     };
 71 
 72 }(this));
