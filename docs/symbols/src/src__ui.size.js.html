1 /**
  2  * @package pklib.ui.size
  3  * @dependence pklib.string
  4  */
  5 (function (global) {
  6     "use strict";
  7 
  8     /**
  9      * @namespace
 10      * @type {Object}
 11      */
 12     var pklib = global.pklib || {},
 13         document = global.document || {};
 14 
 15     /**
 16      * Check ui dimensions
 17      * @namespace
 18      */
 19     pklib.ui.size = {
 20         /**
 21          * @memberOf pklib.ui.size
 22          * @function
 23          * @param {String} name
 24          * @throws {TypeError}
 25          * @returns {Number}
 26          */
 27         window: function (name) {
 28             var clientName;
 29             pklib.common.assert(typeof name === "string", "pklib.ui.size.window: @name: not {String}");
 30 
 31             name = pklib.string.capitalize(name);
 32             clientName = document.documentElement["client" + name];
 33             return (document.compatMode === "CSS1Compat" && clientName) ||
 34                 document.body["client" + name] ||
 35                 clientName;
 36         },
 37         /**
 38          * @memberOf pklib.ui.size
 39          * @function
 40          * @param {String} name
 41          * @returns {Number}
 42          */
 43         document: function (name) {
 44             var clientName,
 45                 scrollBodyName,
 46                 scrollName,
 47                 offsetBodyName,
 48                 offsetName;
 49 
 50             pklib.common.assert(typeof name === "string", "pklib.ui.size.document: @name: not {String}");
 51 
 52             name = pklib.string.capitalize(name);
 53             clientName = document.documentElement["client" + name];
 54             scrollBodyName = document.body["scroll" + name];
 55             scrollName = document.documentElement["scroll" + name];
 56             offsetBodyName = document.body["offset" + name];
 57             offsetName = document.documentElement["offset" + name];
 58             return Math.max(clientName, scrollBodyName, scrollName, offsetBodyName, offsetName);
 59         },
 60         /**
 61          * @memberOf pklib.ui.size
 62          * @function
 63          * @param {HTMLElement} obj
 64          * @param {String} name
 65          * @returns {Number}
 66          */
 67         object: function (obj, name) {
 68             pklib.common.assert(typeof name === "string", "pklib.ui.size.object: @name: not {String}");
 69             pklib.common.assert(pklib.dom.is_node(obj), "pklib.ui.size.object: @obj: not {HTMLElement}");
 70 
 71             name = pklib.string.capitalize(name);
 72             var client = obj["client" + name],
 73                 scroll = obj["scroll" + name],
 74                 offset = obj["offset" + name];
 75             return Math.max(client, scroll, offset);
 76         }
 77     };
 78 
 79 }(this));
