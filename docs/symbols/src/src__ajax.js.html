1 /**
  2  * @package pklib.ajax
  3  * @dependence pklib.array, pklib.common
  4  */
  5 (function (global) {
  6     "use strict";
  7     /** @namespace */
  8     var pklib = global.pklib || {},
  9         /**
 10          * Default time what is timeout to use function pklib.ajax
 11          * @private
 12          * @constant
 13          * @type Number
 14          */
 15         DEFAULT_TIMEOUT_TIME = 30000,
 16         /** @constant */
 17         REQUEST_STATE_UNSENT = 0,
 18         /** @constant */
 19         // REQUEST_STATE_OPENED = 1,
 20         /** @constant */
 21         // REQUEST_STATE_HEADERS_RECEIVED = 2,
 22         /** @constant */
 23         // REQUEST_STATE_LOADING = 3,
 24         /** @constant */
 25         REQUEST_STATE_DONE = 4,
 26         /**
 27          * Array contain key as url, value as ajax response
 28          * @private
 29          * @type Array
 30          */
 31         cache = [],
 32         /**
 33          * Use when state in request is changed or if used cache is handler to request.
 34          * @private
 35          * @function
 36          * @param {Object} settings
 37          * @param {XMLHttpRequest} xhr
 38          */
 39         handler = function (settings, xhr) {
 40             var contentType,
 41                 xmlContentType = ["application/xml", "text/xml"],
 42                 property = "responseText";
 43 
 44             if (xhr.readyState === REQUEST_STATE_DONE && xhr.status !== REQUEST_STATE_UNSENT) {
 45                 if (settings.cache) {
 46                     cache[settings.url] = xhr;
 47                 }
 48 
 49                 contentType = xhr.getResponseHeader("Content-Type");
 50 
 51                 if (pklib.array.in_array(contentType, xmlContentType)) {
 52                     property = "responseXML";
 53                 }
 54 
 55                 settings.done.call(null, xhr[property]);
 56 
 57                 // clear memory
 58                 xhr = null;
 59             }
 60         },
 61         /**
 62          * Handler to unusually situation - timeout.
 63          * @private
 64          * @function
 65          * @param {Object} settings
 66          * @param {XMLHttpRequest} xhr
 67          * @throws {Error} If exists timeout on request
 68          */
 69         timeoutHandler = function (settings, xhr) {
 70             // clear memory
 71             xhr = null;
 72             // throw exception
 73             throw new Error("pklib.ajax.load: timeout on url: " + settings.url);
 74         },
 75         /**
 76          * Method use when request has timeout
 77          * @private
 78          * @function
 79          * @param {Object} settings
 80          * @param {XMLHttpRequest} xhr
 81          * @throws {Error} If exists timeout on request
 82          */
 83         requestTimeout = function (settings, xhr) {
 84             if (typeof xhr.aborted === "undefined" &&
 85                     typeof xhr.error === "undefined" &&
 86                     xhr.readyState === REQUEST_STATE_DONE &&
 87                     xhr.status === REQUEST_STATE_UNSENT) {
 88                 xhr.abort();
 89                 timeoutHandler.call(null, settings, xhr);
 90             }
 91         },
 92         /**
 93          * Try to create Internet Explorer XMLHttpRequest
 94          * @private
 95          * @function
 96          * @throws {Error} If can not create XMLHttpRequest object
 97          * @returns {Object|Undefined} ActiveXObject object
 98          */
 99         create_microsoft_xhr = function () {
100             var xhr;
101             try {
102                 xhr = new global.ActiveXObject("Msxml2.XMLHTTP");
103             } catch (ignore) {
104                 try {
105                     xhr = new global.ActiveXObject("Microsoft.XMLHTTP");
106                 } catch (ignored) {
107                     throw new Error("pklib.ajax.load: cannot create XMLHttpRequest object");
108                 }
109             }
110             return xhr;
111         },
112         /**
113          * Try to create XMLHttpRequest
114          * @private
115          * @function
116          * @throws {Error} If can not create XMLHttpRequest object
117          * @returns {Object|Undefined} XMLHttpRequest object
118          */
119         create_xhr = function () {
120             var xhr;
121             try {
122                 xhr = new global.XMLHttpRequest();
123             } catch (ignore) {
124                 xhr = create_microsoft_xhr();
125             }
126             return xhr;
127         },
128         /**
129          * Add headers to xhr object
130          * @private
131          * @function
132          * @param {Object} settings
133          * @param {XMLHttpRequest} xhr
134          */
135         add_headers_to_xhr = function (settings, xhr) {
136             var header,
137                 headers = settings.headers;
138 
139             if (headers !== null) {
140                 for (header in headers) {
141                     if (headers.hasOwnProperty(header)) {
142                         xhr.setRequestHeader(header, headers[header]);
143                     }
144                 }
145             }
146         },
147         /**
148          * Add timeout service to xhr object
149          * @private
150          * @function
151          * @param {Object} settings
152          * @param {XMLHttpRequest} xhr
153          */
154         add_timeout_service_to_xhr = function (settings, xhr) {
155             if (typeof xhr.ontimeout === "function") {
156                 xhr.ontimeout = timeoutHandler.bind(null, settings, xhr);
157             } else {
158                 pklib.common.defer(requestTimeout.bind(null, settings, xhr), settings.timeout);
159             }
160         },
161         /**
162          * Add error service to xhr object
163          * @private
164          * @function
165          * @param {Object} settings
166          * @param {XMLHttpRequest} xhr
167          */
168         add_error_service_to_xhr = function (settings, xhr) {
169             xhr.onerror = function () {
170                 xhr.error = true;
171                 settings.error.bind(null, settings, xhr);
172             };
173         },
174         /**
175          * Check is response on this request is in cache
176          * @private
177          * @function
178          * @param {Object} settings
179          * @returns {Boolean}
180          */
181         is_response_in_cache = function (settings) {
182             return settings.cache && cache[settings.url];
183         },
184         /**
185          * Return object what is default configuration of request
186          * @private
187          * @function
188          * @returns {Object} Default configuration
189          */
190         set_default_settings = function () {
191             /**
192              * Request settings, contain ex. headers, callback when run after request finish.
193              * Default timeout on request is 30 seconds. This is default timeout from popular web servers
194              * ex. Apache, ngninx.
195              * Default request hasn't any headers.
196              * Default cache is disabled.
197              * Default asynchronous is enable.
198              */
199             return {
200                 type: "get",
201                 async: true,
202                 cache: false,
203                 url: null,
204                 params: null,
205                 timeout: DEFAULT_TIMEOUT_TIME,
206                 headers: {},
207                 /**
208                  * Function run after request ended
209                  * In params exists only: response
210                  */
211                 done: function () {
212                     // do something with response
213                 },
214                 error: function () {
215                     // do something when appear error in request
216                 }
217             };
218         },
219         /**
220          * Check url in request is defined.
221          * Throw error if is undefined
222          * @param {Object} settings
223          * @throws {Error} If unset request url
224          */
225         check_if_url_is_defined = function (settings) {
226             pklib.common.assert(settings.url !== null, "pklib.ajax.load: undefined request url");
227         },
228         /**
229          * Service to send request to server.
230          * With first param, which is hashmap, define params, ex. request url
231          * @namespace
232          */
233         ajax = {
234             /**
235              * Send request to server on url defined in config.url.
236              * Method throw exception when request have timeout on server or if url is not set.
237              * Also, every response (if config.cache is true) saved to hashmap by key config.url.
238              * Method on first try to can create XMLHttpRequest if browser doesn't support, check
239              * if browser support object ActiveXObject which is implemented in Internet Explorer.
240              * @memberOf ajax
241              * @function
242              * @param {Object} config
243              * <pre>
244              * {
245              *      {String} [type="get"]
246              *      {Boolean} [async=true]
247              *      {Boolean} [cache=false]
248              *      {String} url
249              *      {Object} [params]
250              *      {Object} [headers]
251              *      {Function} [done]
252              *      {Function} [error]
253              * }
254              * </pre>
255              * @example
256              * <pre>
257              * pklib.ajax.load({
258              *      type: "post",
259              *      async: false,
260              *      cache:  true,
261              *      url: "http://example.org/check-item.php",
262              *      params: {
263              *          id: 33
264              *      },
265              *      headers: {
266              *          "User-Agent": "tv"
267              *      },
268              *      done: function (res) {
269              *          console.log(res);
270              *      }
271              * });
272              * </pre>
273              * @throws {Error} If unset request url
274              * @returns {XMLHttpRequest|Null}
275              */
276             load: function (config) {
277                 var xhr = null,
278                     settings = set_default_settings();
279 
280                 settings = pklib.object.mixin(settings, config);
281                 settings.type = settings.type.toUpperCase();
282 
283                 check_if_url_is_defined(settings);
284 
285                 if (is_response_in_cache(settings)) {
286                     handler.call(null, settings, cache[settings.url]);
287                 } else {
288                     xhr = create_xhr();
289                     xhr.onreadystatechange = handler.bind(null, settings, xhr);
290                     xhr.open(settings.type, settings.url, settings.async);
291 
292                     add_headers_to_xhr(settings, xhr);
293                     add_timeout_service_to_xhr(settings, xhr);
294                     add_error_service_to_xhr(settings, xhr);
295                     xhr.send(settings.params);
296                 }
297                 return xhr;
298             },
299             /**
300              * Stop request setting in param
301              * @memberOf ajax
302              * @function
303              * @param {XMLHttpRequest} xhr XMLHttpRequest object, or ActiveXObject object if Internet Explorer
304              */
305             stop: function (xhr) {
306                 xhr.abort();
307                 xhr.aborted = true;
308                 // clear memory
309                 xhr = null;
310             }
311         };
312 
313     pklib.ajax = ajax;
314 }(this));
