 51 if (typeof Function.prototype.bind !== "function") {
 52     /**
 53      * Creates a new function that, when called, itself calls this function in the context of the provided this value,
 54      * with a given sequence of arguments preceding any provided when the new function was called.
 55      *
 56      * <pre>
 57      * Method of "Function"
 58      * Implemented in	JavaScript 1.8.5
 59      * ECMAScript Edition	ECMAScript 5th Edition
 60      * </pre>
 61      *
 62      * @param {*} that Context
 63      * @returns {Function}
 64      */
 65     Function.prototype.bind = function (that) {
 66         "use strict";
 67 
 68         var method = this,
 69             slice = Array.prototype.slice,
 70             args = slice.apply(arguments, [1]);
 71 
 72         return function () {
 73             return method.apply(that, args.concat(slice.apply(arguments, [0])));
 74         };
 75     };
 76 }
