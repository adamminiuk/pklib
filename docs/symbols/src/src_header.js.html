 39 if (typeof Function.prototype.bind !== "function") {
 40     /**
 41      * Creates a new function that, when called, itself calls this function in the context of the provided this value,
 42      * with a given sequence of arguments preceding any provided when the new function was called.
 43      *
 44      * <pre>
 45      * Method of "Function"
 46      * Implemented in JavaScript 1.8.5
 47      * ECMAScript Edition ECMAScript 5th Edition
 48      * </pre>
 49      *
 50      * @param {*} that Context
 51      * @returns {Function}
 52      */
 53     Function.prototype.bind = function (that) {
 54         "use strict";
 55 
 56         var method = this,
 57             slice = [].slice,
 58             args = slice.apply(arguments, [1]);
 59 
 60         return function () {
 61             return method.apply(that, args.concat(slice.apply(arguments, [0])));
 62         };
 63     };
 64 }
