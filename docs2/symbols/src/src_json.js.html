1 /**
  2  * @package pklib.json, pklib.string
  3  */
  4 (function (global) {
  5     "use strict";
  6     /** @namespace */
  7     var pklib = global.pklib || {},
  8         /**
  9          * @private
 10          * @function
 11          * @param {Function} fun
 12          */
 13         getFunctionName = function (fun) {
 14             var text = fun.toString().split("\n")[0].replace("function ", "");
 15             return text.substr(0, text.indexOf("(")) + "()";
 16         },
 17         /**
 18          * @private
 19          * @function
 20          * @param {Object} object
 21          */
 22         getLastElement = function (object) {
 23             var i,
 24                 len = 0;
 25             for (i in object) {
 26                 if (object.hasOwnProperty(i)) {
 27                     len += 1;
 28                 }
 29             }
 30             return len;
 31         },
 32         /**
 33          * @private
 34          * @function
 35          * @param {Number} len
 36          */
 37         getIndent = function (len) {
 38             var i,
 39                 fix = "\t",
 40                 source = "";
 41             for (i = 0; i < len; ++i) {
 42                 source += fix;
 43             }
 44             return source;
 45         },
 46         /**
 47          * JSON manager
 48          * @namespace
 49          */
 50         json = {
 51             /**
 52              * @memberOf json
 53              * @function
 54              * @param {Array} object
 55              * @returns {String}
 56              */
 57             stringify: function (object) {
 58                 var source = "",
 59                     args = Array.prototype.slice.call(arguments),
 60                     index = args[1] || 0,
 61                     i,
 62                     item,
 63                     len = 0;
 64 
 65                 switch (true) {
 66                     case typeof object === "undefined":
 67                         // Undefined
 68                         source = undefined;
 69                         break;
 70                     case object === null:
 71                         // Null
 72                         source = null;
 73                         break;
 74                     case typeof object === "boolean":
 75                         // Boolean
 76                         source = object;
 77                         break;
 78                     case typeof object === "number":
 79                         // Number
 80                         source = object;
 81                         break;
 82                     case pklib.string.isString(object):
 83                         // String
 84                         source = '"' + object + '"';
 85                         break;
 86                     case typeof object === "function":
 87                         // Function
 88                         source = getFunctionName(object);
 89                         break;
 90                     case pklib.array.isArray(object):
 91                         // Array
 92                         if (object.length === 0) {
 93                             return "[]";
 94                         }
 95                         source = "[\n" + getIndent(index);
 96                         index += 1;
 97                         len = object.length;
 98                         for (i = 0; i < len; ++i) {
 99                             source += getIndent(index) + pklib.json.stringify(object[i], index);
100                             if (i !== len - 1) {
101                                 source += ",\n";
102                             }
103                         }
104                         index -= 1;
105                         source += "\n" + getIndent(index) + "]";
106                         break;
107                     case pklib.object.isObject(object):
108                         // Object
109                         source = "{\n";
110                         index += 1;
111                         for (item in object) {
112                             if (object.hasOwnProperty(item)) {
113                                 source += getIndent(index) + '"' + item + '": ' + pklib.json.stringify(object[item], index);
114                                 if (item !== getLastElement(object)) {
115                                     source += ",\n";
116                                 }
117                             }
118                         }
119                         index -= 1;
120                         source += "\n" + getIndent(index) + "}";
121                         break;
122                     default:
123                         source = "---";
124                     }
125 
126                 return source;
127             },
128             /**
129              * @memberOf json
130              * @function
131              * @param {Object} source
132              * @param {Boolean} toJson
133              * @throws {TypeError} If first param is not object, second is null
134              * @returns {String}
135              */
136             serialize: function (source, toJson) {
137                 if (typeof source !== "object" || pklib.common.assert(source, null)) {
138                     throw new TypeError("pklib.json.serialize: @source: not object");
139                 }
140 
141                 var amp = false,
142                     item,
143                     value,
144                     mtz,
145                     response = '';
146 
147                 if (toJson) {
148                     response += "{";
149                 }
150                 for (item in source) {
151                     if (source.hasOwnProperty(item)) {
152                         if (amp) {
153                             if (toJson) {
154                                 response += ',';
155                             } else {
156                                 response += '&';
157                             }
158                         } else {
159                             amp = true;
160                         }
161 
162                         value = '';
163                         if (typeof source[item] !== "undefined" && source[item] !== null) {
164                             value = source[item];
165                         }
166 
167                         mtz = toJson ? '"' : '';
168                         response += item;
169                         response += toJson ? ':' : '=';
170                         response += mtz;
171                         response += value + mtz;
172                     }
173                 }
174                 if (toJson) {
175                     response += "}";
176                 }
177 
178                 return response;
179             }
180         };
181 
182     pklib.json = json;
183 }(this));
