1 /**
  2  * @package pklib.aspect
  3  */
  4 (function (global) {
  5     "use strict";
  6 
  7     /** @namespace */
  8     var pklib = global.pklib,
  9 
 10         /**
 11          * Bind function to aspect
 12          * Create method with merge first and second.
 13          * Second method is run after first.
 14          * @param {Function} fun The function to bind aspect function
 15          * @param {Function} asp The aspect function
 16          * @namespace
 17          * @throws {TypeError} If any param is not function
 18          * @returns {Function}
 19          */
 20         aspect = function (fun, asp) {
 21             var that = this;
 22             if (typeof fun !== "function") {
 23                 throw new TypeError("pklib.aspect: @func: not function");
 24             }
 25             if (typeof asp !== "function") {
 26                 throw new TypeError("pklib.aspect: @asp: not function");
 27             }
 28             return function () {
 29                 asp.call(that);
 30                 return fun.apply(that, arguments);
 31             };
 32         };
 33 
 34     pklib.aspect = aspect;
 35 }(this));
