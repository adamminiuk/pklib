1 /**
  2  * @package pklib.browser
  3  */
  4 (function (global) {
  5     "use strict";
  6 
  7     /** @namespace */
  8     var pklib = global.pklib || {},
  9 
 10         /**
 11          * Array with browsers name
 12          */
 13         browsers = ["msie", "chrome", "safari", "opera", "mozilla", "konqueror"],
 14 
 15         /**
 16          * Get best information about browser
 17          * @namespace
 18          */
 19         browser = {
 20             /**
 21              * Get browser name by checking userAgent in global object navigator.
 22              * @returns {String}
 23              */
 24             getName: function () {
 25                 var i, browser,
 26                     len = browsers.length,
 27                     userAgent = navigator.userAgent.toLowerCase();
 28 
 29                 for (i = 0; i < len; ++i) {
 30                     browser = browsers[i];
 31                     if (new RegExp(browser).test(userAgent)) {
 32                         return browser;
 33                     }
 34                 }
 35                 return "undefined";
 36             },
 37             /**
 38              * Get browser version by checking userAgent.
 39              * Parse userAgent to find next 3 characters.
 40              * @returns {String}
 41              */
 42             getVersion: function () {
 43                 var i, len = browsers.length, browser, cur,
 44                     userAgent = navigator.userAgent.toLowerCase();
 45 
 46                 for (i = 0; i < len; ++i) {
 47                     browser = browsers[i];
 48                     cur = userAgent.indexOf(browser);
 49                     if (cur !== -1) {
 50                         return userAgent.substr(cur + len + 1, 3);
 51                     }
 52                 }
 53                 return "-1";
 54             }
 55         };
 56 
 57     pklib.browser = browser;
 58 }(this));
