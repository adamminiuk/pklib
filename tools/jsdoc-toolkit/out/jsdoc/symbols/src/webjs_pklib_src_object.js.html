1 /**
  2  * @package pklib.object
  3  */
  4 (function (global) {
  5     "use strict";
  6 
  7     /** @namespace */
  8     var pklib = global.pklib || {},
  9 
 10         /**
 11          * Module to service object
 12          * @namespace
 13          */
 14         object =  {
 15             /**
 16              * Check if is object
 17              * @param {Object} o
 18              * @returns {Boolean}
 19              */
 20             isObject: function (o) {
 21                 return o && typeof o === "object" &&
 22                     typeof o.hasOwnProperty === "function" &&
 23                     typeof o.isPrototypeOf === "function" &&
 24                     typeof o.length === "undefined";
 25             },
 26             /**
 27              * @param {Array|Object} target
 28              * @param {Array|Object} source
 29              * @returns {Array}
 30              */
 31             mixin: function (target, source) {
 32                 var i, len = 0, element, item;
 33 
 34                 if (pklib.array.isArray(target) && pklib.array.isArray(source)) {
 35                     len = source.length;
 36                     for (i = 0; i < len; ++i) {
 37                         element = source[i];
 38                         if (!pklib.array.inArray(element, target)) {
 39                             target.push(element);
 40                         }
 41                     }
 42                     target.sort();
 43                     return target;
 44                 } else {
 45                     for (item in source) {
 46                         if (source.hasOwnProperty(item)) {
 47                             if (pklib.object.isObject(target[item])) {
 48                                 target[item] = pklib.object.mixin(target[item], source[item]);
 49                             } else {
 50                                 target[item] = source[item];
 51                             }
 52                         }
 53                     }
 54                     return target;
 55                 }
 56             }
 57         };
 58 
 59     pklib.object = object;
 60 }(this));
