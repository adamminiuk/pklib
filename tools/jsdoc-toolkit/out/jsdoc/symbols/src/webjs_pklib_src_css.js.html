1 /**
  2  * @package pklib.css
  3  * @dependence pklib.string. pklib.dom
  4  */
  5 (function (global) {
  6     "use strict";
  7 
  8     /** @namespace */
  9     var pklib = global.pklib || {},
 10 
 11         /**
 12          * RegExp use to delete white chars
 13          */
 14         rclass = /[\n\t\r]/g,
 15 
 16         /**
 17          * Check typeof params
 18          * @param {String} cssClass
 19          * @param {HTMLElement} element
 20          * @throws {TypeError} If first param is not string, or second param is not Node
 21          */
 22         checkParams = function (cssClass, element, callFuncName) {
 23             var prefix = "pklib.css." + callFuncName;
 24             if (typeof cssClass !== "string") {
 25                 throw new TypeError(prefix + ": @cssClass: not String");
 26             }
 27             if (!pklib.dom.isNode(element)) {
 28                 throw new TypeError(prefix + ": @element: not HTMLElement");
 29             }
 30         },
 31 
 32         /**
 33          * Utils method related css on tags in DOM tree
 34          * @namespace
 35          */
 36         css = {
 37             /**
 38              * Add CSS class to element define in second parameter.
 39              * @param {String} cssClass
 40              * @param {HTMLElement} element
 41              * @throws {TypeError} If first param is not string, or second param is not Node
 42              */
 43             addClass: function (cssClass, element) {
 44                 checkParams(cssClass, element, "addClass");
 45                 var classElement = element.className;
 46                 if (!pklib.css.hasClass(cssClass, element)) {
 47                     if (classElement.length) {
 48                         classElement += " " + cssClass;
 49                     } else {
 50                         classElement = cssClass;
 51                     }
 52                 }
 53                 element.className = classElement;
 54             },
 55             /**
 56              * Remove CSS class from element define in second parameter.
 57              * @param {String} cssClass
 58              * @param {HTMLElement} element
 59              * @throws {TypeError} If first param is not string, or second param is not Node
 60              */
 61             removeClass: function (cssClass, element) {
 62                 checkParams(cssClass, element, "removeClass");
 63                 var regexp = new RegExp("(\\s" + cssClass + ")|(" + cssClass + "\\s)|" + cssClass, "i");
 64                 element.className = pklib.string.trim(element.className.replace(regexp, ""));
 65             },
 66             /**
 67              * Check if element has CSS class
 68              * @param {String} cssClass
 69              * @param {HTMLElement} element
 70              * @throws {TypeError} If first param is not string, or second param is not Node
 71              * @returns {Boolean}
 72              */
 73             hasClass: function (cssClass, element) {
 74                 checkParams(cssClass, element, "hasClass");
 75                 var className = " " + cssClass + " ";
 76                 return ((" " + element.className + " ").replace(rclass, " ").indexOf(className) > -1);
 77             }
 78         };
 79 
 80     pklib.css = css;
 81 }(this));
