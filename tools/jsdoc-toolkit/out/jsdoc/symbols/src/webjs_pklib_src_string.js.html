1 /**
  2  * @package pklib.string
  3  */
  4 (function (global) {
  5     "use strict";
  6     var pklib = global.pklib || {},
  7 
  8         /**
  9          * String service manager
 10          * @namespace
 11          */
 12         string = {
 13             /**
 14              * @param source {String}
 15              * @returns {Boolean}
 16              */
 17             isString: function (source) {
 18                 return typeof source === "string";
 19             },
 20             /**
 21              * @param source {String}
 22              * @returns {Boolean}
 23              */
 24             isLetter: function (source) {
 25                 return pklib.string.isString(source) && /^[a-zA-Z]$/.test(source);
 26             },
 27             /**
 28              * @param source {String}
 29              * @returns {String}
 30              */
 31             trim: function (source) {
 32                 return source.replace(/^\s+|\s+$/g, "");
 33             },
 34             /**
 35              * @param source {String}
 36              * @returns {String}
 37              */
 38             slug: function (source) {
 39                 var result = source.toLowerCase().replace(/\s/mg, "-");
 40                 result = result.replace(/[^a-zA-Z0-9\-]/mg, function (ch) {
 41                     switch (ch.charCodeAt(0)) {
 42                     case 261:
 43                         return String.fromCharCode(97);
 44                     case 281:
 45                         return String.fromCharCode(101);
 46                     case 243:
 47                         return String.fromCharCode(111);
 48                     case 347:
 49                         return String.fromCharCode(115);
 50                     case 322:
 51                         return String.fromCharCode(108);
 52                     case 378:
 53                     case 380:
 54                         return String.fromCharCode(122);
 55                     case 263:
 56                         return String.fromCharCode(99);
 57                     case 324:
 58                         return String.fromCharCode(110);
 59 
 60                     default:
 61                         return "";
 62                     }
 63                 });
 64                 return result;
 65             },
 66             /**
 67              * @param source {String}
 68              * @returns {String}
 69              */
 70             capitalize: function (source) {
 71                 return source.substr(0, 1).toUpperCase() + source.substring(1, source.length).toLowerCase();
 72             },
 73             /**
 74              * @param source {String}
 75              * @returns {String}
 76              */
 77             delimiterSeparatedWords: function (source) {
 78                 return source.replace(/[A-ZĘÓĄŚŁŻŹĆŃ]/g, function (match) {
 79                     return "-" + match.toLowerCase();
 80                 });
 81             },
 82             /**
 83              * @param source {String}
 84              * @returns {String}
 85              */
 86             stripTags: function (source) {
 87                 if (source && source.length !== 0) {
 88                     return source.replace(/\\<\\S\\>/g, "");
 89                 }
 90                 return source;
 91             },
 92             /**
 93              * @param source {String}
 94              * @returns {String}
 95              */
 96             camelCase: function (source) {
 97                 while (source.indexOf("-") !== -1) {
 98                     var pos = source.indexOf("-"),
 99                         pre = source.substr(0, pos),
100                         sub = source.substr(pos + 1, 1).toUpperCase(),
101                         post = source.substring(pos + 2, source.length);
102                     source = pre + sub + post;
103                 }
104                 return source;
105             },
106             /**
107              * @param source {String}
108              * @param length {Number}
109              * @returns {String}
110              */
111             slice: function (source, length) {
112                 var item,
113                     text = "",
114                     num = source.length;
115 
116                 for (item = 0; item < num; item += 1) {
117                     text += source.substr(item, 1);
118                     if (item === length - 1) {
119                         if (num - length >= 1) {
120                             text += "...";
121                         }
122                         break;
123                     }
124                 }
125                 return text;
126             }
127         };
128 
129     pklib.string = string;
130 }(this));
