1 /**
  2  * @package pklib.utils
  3  * @dependence pklib.common, pklib.dom, pklib.event
  4  */
  5 (function (global) {
  6     "use strict";
  7 
  8     /** @namespace */
  9     var pklib = global.pklib || {},
 10         document = global.document || {},
 11 
 12         /**
 13          * Utils tools
 14          * @namespace
 15          */
 16         utils = {
 17             ascii: {
 18                 letters: {
 19                     lower: [113, 119, 101, 114, 116, 121, 117, 105, 111, 112, 97, 115, 100, 102, 103, 104, 106, 107, 108, 122, 120, 99, 118, 98, 110, 109],
 20                     upper: [81, 87, 69, 82, 84, 89, 85, 73, 79, 80, 65, 83, 68, 70, 71, 72, 74, 75, 76, 90, 88, 67, 86, 66, 78, 77]
 21                 }
 22             },
 23             action: {
 24                 /**
 25                  * @param {HTMLElement} obj
 26                  */
 27                 clearfocus: function (obj) {
 28                     if (pklib.dom.isElement(obj)) {
 29                         pklib.event.add(obj, "focus", function () {
 30                             if (obj.value === obj.defaultValue) {
 31                                 obj.value = "";
 32                             }
 33                         });
 34                         pklib.event.add(obj, "blur", function () {
 35                             if (obj.value === "") {
 36                                 obj.value = obj.defaultValue;
 37                             }
 38                         });
 39                     }
 40                 },
 41                 /**
 42                  * @param {HTMLElement} area
 43                  */
 44                 outerlink: function (area) {
 45                     area = area || document;
 46                     var i,
 47                         link,
 48                         links = pklib.dom.byTag("a", area),
 49                         len = links.length;
 50 
 51                     function opentrigger(evt) {
 52                         global.open(this.href);
 53                         evt.preventDefault();
 54                     }
 55 
 56                     for (i = 0; i < len; ++i) {
 57                         link = links[i];
 58                         if (link.rel === "outerlink") {
 59                             pklib.event.add(link, "click", opentrigger.bind(link));
 60                         }
 61                     }
 62                 },
 63                 /**
 64                  * @param {HTMLElement} element
 65                  * @param {String} text
 66                  */
 67                 confirm: function (element, text) {
 68                     var response;
 69                     if (typeof element !== "undefined") {
 70                         text = text || "Sure?";
 71 
 72                         pklib.event.add(element, "click", function (evt) {
 73                             response = global.confirm(text);
 74                             if (pklib.common.assert(response, true)) {
 75                                 return true;
 76                             } else {
 77                                 evt.preventDefault();
 78                             }
 79                         });
 80                     }
 81                 }
 82             }
 83         };
 84 
 85     pklib.utils = utils;
 86 }(this));
