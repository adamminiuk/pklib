1 /**
  2  * @package pklib.url
  3  */
  4 (function (global) {
  5     "use strict";
  6 
  7     /** @namespace */
  8     var pklib = global.pklib || {},
  9 
 10         /**
 11          * Document.location object
 12          */
 13         loc = global.location,
 14 
 15         /**
 16          * Url helper manager
 17          * @namespace
 18          */
 19         url = {
 20             /**
 21              * @returns {String}
 22              */
 23             getProtocol: function () {
 24                 return loc.protocol;
 25             },
 26             /**
 27              * @returns {String}
 28              */
 29             getHost: function () {
 30                 return loc.host;
 31             },
 32             /**
 33              * @returns {String}
 34              */
 35             getPort: function () {
 36                 return loc.port || 80;
 37             },
 38             /**
 39              * @returns {String}
 40              */
 41             getUri: function () {
 42                 return loc.pathname;
 43             },
 44             /**
 45              * @returns {Array}
 46              */
 47             getParams: function () {
 48                 var i,
 49                     item,
 50                     len,
 51                     params = loc.search,
 52                     paramsList = {};
 53                 if (params.substr(0, 1) === "?") {
 54                     params = params.substr(1);
 55                 }
 56                 params = params.split("&");
 57                 len = params.length;
 58                 for (i = 0; i < len; ++i) {
 59                     item = params[i].split("=");
 60                     paramsList[item[0]] = item[1];
 61                 }
 62                 return paramsList;
 63             },
 64             /**
 65              * @param {String} key
 66              * @returns {String}
 67              */
 68             getParam: function (key) {
 69                 var params = loc.search,
 70                     i,
 71                     item,
 72                     len = params.length;
 73                 if (params.substr(0, 1) === "?") {
 74                     params = params.substr(1);
 75                 }
 76                 params = params.split("&");
 77                 for (i = 0; i < len; ++i) {
 78                     item = params[i].split("=");
 79                     if (item[0] === key) {
 80                         return item[1];
 81                     }
 82                 }
 83                 return null;
 84             },
 85             /**
 86              * @returns {String}
 87              */
 88             getHash: function () {
 89                 return loc.hash;
 90             }
 91         };
 92 
 93     pklib.url = url;
 94 }(this));
