1 /**
  2  * @package pklib.ui
  3  * @dependence pklib.string. pklib.dom
  4  */
  5 (function (global) {
  6     "use strict";
  7 
  8     /** @namespace */
  9     var pklib = global.pklib || {},
 10         document = global.document,
 11 
 12         /**
 13          * User Interface
 14          * @namespace
 15          */
 16         ui = {
 17             /**
 18              * @param {HTMLElement} element
 19              * @param {HTMLElement} wrapper
 20              * @throws {TypeError} If first param is not HTMLElement
 21              * @returns {Array}
 22              */
 23             center: function (element, wrapper) {
 24                 var left = null,
 25                     top = null,
 26                     pus = this.size;
 27 
 28                 if (!pklib.dom.isElement(element)) {
 29                     throw new TypeError("pklib.ui.center: @element: not Element");
 30                 }
 31 
 32                 if (wrapper === document.body) {
 33                     left = (Math.max(pus.window("width"), pus.document("width")) - pus.object(element, "width")) / 2;
 34                     top = (Math.max(pus.window("height"), pus.document("height")) - pus.object(element, "height")) / 2;
 35                 } else {
 36                     left = (pus.window("width") - pus.object(element, "width")) / 2;
 37                     top = (pus.window("height") - pus.object(element, "height")) / 2;
 38                 }
 39                 element.style.left = left + "px";
 40                 element.style.top = top + "px";
 41                 element.style.position = "absolute";
 42 
 43                 return [left, top];
 44             },
 45             /**
 46              * @param {HTMLElement} element
 47              * @param {HTMLElement} wrapper
 48              * @returns {Array}
 49              */
 50             maximize: function (element, wrapper) {
 51                 var width = null,
 52                     height = null,
 53                     pus = pklib.ui.size;
 54 
 55                 if (wrapper === document.body) {
 56                     width = Math.max(pus.window("width"), pus.document("width"));
 57                     height = Math.max(pus.window("height"), pus.document("height"));
 58                     if (pklib.browser.getName() === "msie") {
 59                         width -= 20;
 60                     }
 61                 } else {
 62                     width = pus.object(wrapper, "width");
 63                     height = pus.object(wrapper, "height");
 64                 }
 65                 element.style.width = width;
 66                 element.style.height = height;
 67                 return [width, height];
 68             },
 69             /**
 70              * @param {Number} param
 71              * @param {Boolean} animate
 72              */
 73             scrollTo: function (param, animate) {
 74                 var interval = null;
 75                 if (pklib.common.assert(animate, true)) {
 76                     interval = setInterval(function () {
 77                         document.body.scrollTop -= 5;
 78                         if (document.body.scrollTop <= 0) {
 79                             clearInterval(interval);
 80                         }
 81                     }, 1);
 82                 } else {
 83                     document.body.scrollTop = param + "px";
 84                 }
 85             }
 86         };
 87 
 88     pklib.ui = ui;
 89 }(this));
