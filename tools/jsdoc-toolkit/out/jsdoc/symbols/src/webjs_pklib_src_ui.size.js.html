1 /**
  2  * @package pklib.ui.size
  3  * @dependence pklib.string
  4  */
  5 (function (global) {
  6     "use strict";
  7 
  8     /** @namespace */
  9     var pklib = global.pklib || {},
 10         document = global.document,
 11 
 12         /**
 13          * Check ui dimensions
 14          * @namespace
 15          */
 16         size = {
 17             /**
 18              * @param name {String}
 19              * @throws {TypeError}
 20              * @returns {Number}
 21              */
 22             window: function (name) {
 23                 var clientName;
 24                 if (typeof name === "undefined") {
 25                     throw new TypeError("pklib.ui.size.window: @name: undefined");
 26                 }
 27                 name = pklib.string.capitalize(name);
 28                 clientName = document.documentElement["client" + name];
 29                 return (document.compatMode === "CSS1Compat" && clientName) ||
 30                     document.body["client" + name] ||
 31                     clientName;
 32             },
 33             /**
 34              * @param name {String}
 35              * @returns {Number}
 36              */
 37             document: function (name) {
 38                 var clientName,
 39                     scrollBodyName,
 40                     scrollName,
 41                     offsetBodyName,
 42                     offsetName;
 43                 if (typeof name === "undefined") {
 44                     throw new TypeError("pklib.ui.size.document: @name: undefined");
 45                 }
 46                 name = pklib.string.capitalize(name);
 47                 clientName = document.documentElement["client" + name];
 48                 scrollBodyName = document.body["scroll" + name];
 49                 scrollName = document.documentElement["scroll" + name];
 50                 offsetBodyName = document.body["offset" + name];
 51                 offsetName = document.documentElement["offset" + name];
 52                 return Math.max(clientName, scrollBodyName, scrollName, offsetBodyName, offsetName);
 53             },
 54             /**
 55              * @param obj {HTMLElement}
 56              * @param name {String}
 57              * @returns {number}
 58              */
 59             object: function (obj, name) {
 60                 if (typeof name === "undefined" || typeof obj === "undefined") {
 61                     throw new TypeError("pklib.ui.size.object: @name: undefined");
 62                 }
 63                 name = pklib.string.capitalize(name);
 64                 var client = obj["client" + name], scroll = obj["scroll" + name], offset = obj["offset" + name];
 65                 return Math.max(client, scroll, offset);
 66             }
 67         };
 68 
 69     pklib.ui = global.pklib.ui || {};
 70     pklib.ui.size = size;
 71 }(this));
